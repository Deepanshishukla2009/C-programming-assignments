Q10: Write a program to input time in seconds and convert it to hours:minutes:seconds format.

/*
Sample Test Cases:
Input 1:
3661
Output 1:
1:1:1

Input 2:
7322
Output 2:
2:2:2

*/


#include<stdio.h>
#include<conio.h>
int main()
{
    int time,hour,minute,second;
    printf("enter time in seconds:");
    scanf("%d",&time);
    hour=time/3600;
    minute=(time%3600)/60;
    second=(time%60);
    printf("time= %2d:%2d:%2d\n", hour, minute, second);
    return 0;
}
