Q34: Write a program to check if a number is prime.

/*
Sample Test Cases:
Input 1:
7
Output 1:
Prime

Input 2:
10
Output 2:
Not prime

*/

#include<stdio.h>
#include<conio.h>
int main()
{
    int n,temp=1,i;
    printf("enter the number:");
    scanf("%d",&n);
    if(n<=1)
    {
        temp=0;
    }
    else
    {
        for(i=2;i*i<=n;i++)
        {
            if(n%i==0)
            {
                temp=0;
                break;
            }
        }
    }
    if(temp){
    printf("prime\n");
    }
    else{
    printf("not prime");
    }
    return 0;
}
