Q29: Write a program to calculate the factorial of a number.

/*
Sample Test Cases:
Input 1:
5
Output 1:
120

Input 2:
3
Output 2:
6

*/

#include<stdio.h>
#include<conio.h>
int main()
{
    int n,i,fact=1;
    printf("enter value of n:");
    scanf("%d",&n);
    if(n<0)
    {
        printf("factorial is not defined for negative integer");
    }
    else
    {
        for (i = 1; i <= n; i++)
        {
            fact *= i;
        }
        printf("Factorial of %d is: %d\n", n, fact);
    }
    return 0;
}
