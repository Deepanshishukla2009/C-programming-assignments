Q28: Write a program to print the product of even numbers from 1 to n.

/*
Sample Test Cases:
Input 1:
4
Output 1:
8 (2 * 4)

Input 2:
6
Output 2:
48 (2 * 4 * 6)

*/

#include<stdio.h>
#include<conio.h>
int main()
{
    int n,i;
    long long product=1;
    printf("enter value of n:");
    scanf("%d",&n);
     for (i = 2; i <= n; i += 2)
      {
        product *= i;
    }
    if (n < 2)
    {
        printf("No even numbers between 1 and %d.\n", n);
    }
    else
    {
        printf("Product of even numbers from 1 to %d is: %lld\n", n, product);
    }
    return 0;
}
